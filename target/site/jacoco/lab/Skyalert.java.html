package lab;

import java.io.IOException;
import java.util.Scanner;
import java.util.Vector;
import java.util.logging.Level;
import java.util.logging.Logger;




public class
Skyalert {
    static final Logger logger = Logger.getLogger(Skyalert.class.getName());

    static Vector<Tarifa> tarifas = new Vector<Tarifa>();

    public static String valueOf(Object obj) {
        return (obj == null) ? "null" : obj.toString();
    }

    public static void createTarifas(){
        for(int i = 1;i < 7; i++){
            Tarifa tarifa = new Tarifa();
            tarifa.duracion = i * 4;
            tarifa.precio = i * 1000;
            tarifa.fecha = "02/04/2022";
            tarifa.agencia = "LATAM";
            tarifa.numDeVuelo = "A00"+valueOf(i-1);
            tarifas.addElement(tarifa);
        }
    }

    public static String buscarTarifa(int precio, int duracion){
        createTarifas();
        for(int i=1; i<7; i++){
            if(i*4 == duracion && i*1000 == precio){
                return tarifas.get(i).agencia+" "+ tarifas.get(i).numDeVuelo;
            }
        }
        return "No encontró";
    }

    public static void main(String[] args) throws IOException, InterruptedException {
        logger.info("Determinar Precio y duración:");
        var sc = new Scanner(System.in);
        var precio = sc.nextInt();
        var duracion = sc.nextInt();
        if (logger.isLoggable(Level.INFO)){
            logger.info(buscarTarifa(precio,duracion));
        }
        //Precios con duraciones disponibles ({1000, 4}, {2000, 8}, {3000, 12}, {4000, 16} y {5000, 20})
    }

}
